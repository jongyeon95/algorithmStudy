#include <stdio.h>

int main (){
int a[4];
int b[4];
int c[4];
int ra=0;
int rb=0;
int rc=0;
char ca='A';
char cb='A';
char cc='A';

for(int i=0;i<4;i++){
    scanf("%d",&a[i]);
    if(a[i]==1){
        ra++;
    }
}
for(int i=0;i<4;i++){
    scanf("%d",&b[i]);
    if(b[i]==1){
        rb++;
    }
}
for(int i=0;i<4;i++){
    scanf("%d",&c[i]);
    if(c[i]==1){
        rc++;
    }
}

if(ra==3){ca='A';}
else if(ra==2){ca='B';}
else if(ra==1){ca='C';}
else if(ra==0){ca='D';}
else if(ra==4){ca='E';}
if(rb==3){cb='A';}
else if(rb==2){cb='B';}
else if(rb==1){cb='C';}
else if(rb==0){cb='D';}
else if(rb==4){cb='E';}
if(rc==3){cc='A';}
else if(rc==2){cc='B';}
else if(rc==1){cc='C';}
else if(rc==0){cc='D';}
else if(rc==4){cc='E';}

printf("%c\n%c\n%c\n",ca,cb,cc);

}
